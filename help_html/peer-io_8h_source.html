155 static inline bool
156 tr_isPeerIo (const tr_peerIo * io)
157 {
158  return (io != NULL)
159  && (io->magicNumber == PEER_IO_MAGIC_NUMBER)
160  && (io->refCount >= 0)
161  && (tr_isBandwidth (&io->bandwidth))
162  && (tr_address_is_valid (&io->addr));
163 }
